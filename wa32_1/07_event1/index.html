メッセージ表示：showMessage()

#result に与えられたテキストを表示する共通関数を作成 

      let result = $('#result');

      function showMessage(message) {
          result.text(message);
      }

click: ボタン1, 2, 3


      $('#btn1').on('click', function () {
          showMessage('ボタン1がクリックされました。');
      });

      $('#btn2').on('click', function () {
          showMessage('ボタン2がクリックされました。');
      });

      $('#btn3').on('click', function () {
          showMessage('ボタン3がクリックされました。');
      });

click: ボタン4


      $('#btn4').on('click',
          {
              msg1: 'ボタン4がクリックされました。',
              msg2: 'こんにちは',
              msg3: 'さようなら'
          },
          function (e) {
              let message = e.data.msg1 + ' ' + e.data.msg2  + ' ' +  e.data.msg3;
              showMessage(message);
          }
      );
