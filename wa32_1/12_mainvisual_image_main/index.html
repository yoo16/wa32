次の画像インデックス番号を取得


        current = ++current % items.length
  1. current を1増加
  2. items（li 要素配列）の個数を取得
  3. 余りを求め、current に代入する

画像が5個の場合

current = 0 ( 0 % 5 )
current = 4 ( 1 % 5 )
current = 3 ( 2 % 5 )
current = 2 ( 3 % 5 )
current = 1 ( 4 % 5 )
current = 0 ( 5 % 5 )

メインビジュアル：サンプル


    $(function () {
        function photoChange(target) {
            let items = target.find('li');  // li要素のセレクタを格納した配列
            let current = 0;                // 現在表示されているインデックス
            let easing = 'swing';
            let duration = 1200;
            let showTime = 2000
    
            function open() {
                $(items[current]).fadeIn(
                    duration,
                    easing,
                    function () { setTimeout(change, showTime); }
                );
            }
    
            function close() {
                $(items[current]).fadeOut(duration, easing);
            }
    
            function change() {
                close();
                current = ++current % items.length;
                open();
            }
    
            open();
        }
    
        photoChange($('#photolist'));
      });
