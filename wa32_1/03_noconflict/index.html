main1.js

prototype.js でエラー


    $(function () {
        alert($('body').attr('id'));
    });
    
    document.observe("dom:loaded", function () {
        $("myBody").addClassName("myClass");
    });

main2.js

noConflict() で実行可能


    jQuery.noConflict();

    jQuery(function () {
        alert(jQuery('body').attr('id'));
    });
    
    document.observe("dom:loaded", function () {
        $("myBody").addClassName("myClass");
    });

main3.js

$ の受け渡しが可能


    jQuery.noConflict();

    jQuery(function ($) {
        alert($('body').attr('id'));
    });
    
    document.observe("dom:loaded", function () {
        $("myBody").addClassName("myClass");
    });
