無名関数

無名関数はその名の通り名前のない関数です。 無名関数自体を変数に代入し、実行することができます。 


      let 変数 = function (引数) {
          //処理
      }

無名関数を変数定義して実行


      let calculate = function (a, b) {
          return a + b;
      }
      calculate(3, 4);

submit イベント

Form の Submit ボタンのイベントを処理します。

      $(セレクタ).on('submit', function (event) {

      });

keypress イベント

キーボードのイベントを処理します。 イベント変数でキーボードのキーを検出できます。 


      $(セレクタ).on('keypress', function (event) {
          console.log(event.keyCode);
      });

初期設定の実装


      let init = function () {
          // submit イベントの登録
          target.on('submit', function () {
              check();
              return false;
          });

          // items 配列に要素を入れる
          items = [
              target.find('input[name=forName]'),  //$(‘#myForm') から input name=forName を探す
          ];

          // items を繰り返す（input の 名前、ふりがな）
          $.each(items, function (index) {
              items[index].prop('isSuccess', false);  //items のプロパティ isSuccess を false にする
          });

          // enterキーでsubmit 防止
          target.find('input[type=text]').on({  //input type=text 上でキーが押されたら
              'keypress': function (e) {
                  if (e.keyCode == 13) return false;  enterキーだったら止める
              }
          });
      }

入力チェック

入力チェック


      let check = function () {
          checkEmptyText(items[0], '名前を入力してください');  //items[0] : 初期設定した名前の input要素
          checkEmptyText(items[1], 'ふりがなを入力してください');  //items[1] : 初期設定したふりがなの input要素
      }

未入力チェック


      let checkEmptyText = function (selector, message) {
          if (selector.val() == '') {
              // エラーメッセージを表示
              addErrorMessage(selector, message);
              selector.prop('isSuccess', false);  //isSuccess = false
          } else {
              // エラーメッセージを非表示
              removeErrorMessage(selector);
              selector.prop('isSuccess', true);  //isSuccess = true
          }
      }

before()

親要素の前に要素追加


      $(‘セレクタ’).before(文字列);

input は閉じタグがないため append() でなく before() を利用

エラーメッセージ

エラーメッセージ表示


      let addErrorMessage = function (selector, message) {
          removeErrorMessage(selector);
          let span = $('').addClass('error').html(message);
          selector.before(span);
          selector.addClass('errorInput');
      }

エラーメッセージ非表示


      let msgSelector = selector.parent().find('.error');  //同じ階層で class=error の要素を探す
      if (msgSelector.length != 0) {            //要素が存在すれば
          msgSelector.remove();                 //要素を削除
          selector.removeClass('errorInput');   //class="error" を削除
      }
